\question[5] \ce{HCl} is an acid. Which of the following happens when \ce{HCl} is dissolved in water?
 \begin{choices}
\choice \ce{HCl (aq) + H2O (l) -> H3O^{+} (aq) + Cl^{-}}
\choice \ce{HCl (aq) + H2O (l) -> H3O^{+} (aq) + Cl^{2-}}
\choice \ce{HCl (aq) + H2O (l) -> H3O^{2+} (aq) + 2Cl^{-}}
\choice \ce{HCl (aq) + H2O (l) -> H3O^{+} (aq) + OH^{-}}
\end{choices}

\begin{solution}
A) \ce{HCl (aq) + H2O (l) -> H3O^{+} (aq) + Cl^{-}}
\end{solution}

\vspace{0.2in}

\question[5] \ce{NaOH} is a base. Which of the following happens when \ce{NaOH} is dissolved in water?

 \begin{choices}
\choice \ce{NaOH (aq) -> Na^{+} (aq) + OH^{-} (aq)} 
\choice \ce{NaOH (aq) -> Na^{2+} (aq) + OH^{-} (aq)} 
\choice \ce{NaOH (aq) -> Na^{+} (aq) + OH^{2-} (aq)} 
\choice \ce{NaOH (aq) -> Na^{-} (aq) + H3O^{+} } 
\end{choices}

\begin{solution}
A) \ce{NaOH (aq) -> Na^{+} (aq) + OH^{-} (aq)} 
\end{solution}

 \begin{choices}

\question[5] Which of the following acid/base reactions is correct?

 \begin{choices}
\choice \ce{H2SO4 (aq) + HCl (aq) -> Cl2SO4 (aq) + H2O (l)} 
\choice \ce{HCl (aq) + NaOH (aq) -> NaCl (aq) + H2O (l)} 
\choice \ce{H3PO4 (aq) + HCl (aq) -> ClPO4^{4-} (aq) + H2O (l)} 
\choice \ce{H2CO3 (aq) + NaOH (aq) -> Na2CO3 (aq) + 2H2O (l)} 
\end{choices}

\begin{solution}
B)  \ce{HCl (aq) + NaOH (aq) -> NaCl (aq) + H2O (l)} 
\end{solution}
