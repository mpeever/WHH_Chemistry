\question[5] The enthalpy change of some chemical reaction is negative  ($\Delta H < 0$) . 
Is the reaction endothermic or exothermic?

\begin{choices}
\choice endothermic
\choice exothermic
\end{choices}

\begin{solution}
B) exothermic
\end{solution}

\vspace{0.2in}

\question[5] Which of the following would we expect to have the highest entropy?

\begin{choices}
\choice $1.00 mol$ of \ce{Fe (s)}
\choice $1.00 mol$ of \ce{H2O (l)}
\choice $1.00 mol$ of \ce{OH- (aq)}
\choice $1.00 mol$ of \ce{Cl2 (g)}
\end{choices}

\begin{solution}
D) $1.00 mol$ of \ce{Cl2 (g)}
\end{solution}

\vspace{0.2in}

\question[5] True or False: the entropy of the universe is constantly decreasing.

\begin{choices}
\choice false
\choice true
\end{choices}

\begin{solution}
A) false
\end{solution}

\question[5] The enthalpy change of a given chemical reaction is negative ($\Delta H < 0$) , and its entropy change is positive ($\Delta S > 0$).
Would we expect the $\Delta G$ value for the reaction to be positive or negative?

\begin{choices}
\choice positive ($\Delta G > 0$)
\choice negative ($\Delta G < 0$)
\end{choices}

\begin{solution}
B) negative
\end{solution}
