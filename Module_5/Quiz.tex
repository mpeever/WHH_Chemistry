\question[5] Caroline measures out $1.000$ mole of Hydrogen (\ce{H}) atoms.
If the mass of one \ce{H} atom is $1.00794 amu$, how many grams of \ce{H} does she measure out?
\begin{solution}
\begin{equation} 
\begin{split}
    m &= n \times M \\
        &= (1.000 mol) (1.000794 \frac{g}{mol}) \\
        &= 1.00794 g \\
        &= 1.008 g                                    
 \end{split}
 \end{equation}
 \end{solution}

\question[5] Am\'elie measures out $15.9994 g$ of Oxygen (\ce{O2}). 
If the mass of one \ce{O} atom is $15.9994 amu$, how many moles does she measure out? 
\begin{solution}
\begin{equation} 
\begin{split}
    m &= n \times M \\
    n &= \frac{m}{M} \\
       &= \frac{15.9994 g}{15.9994 \frac{g}{mol}} \\
       &= 1.00000 mol                                 
 \end{split}
 \end{equation}
 \end{solution}
 
\question[5] Gideon has $88 g$ of carbon dioxide (\ce{CO2}).
The atomic mass of Carbon (\ce{C}) is $12.0107 amu$ and the atomic mass of Oxygen (\ce{O}) is $15.9994 amu$.
How many moles of \ce{CO2} does Gideon have?
\begin{solution}
\begin{equation} 
\begin{split}
    M &= 1 \times M_{C} + 2 \times M_{O} \\
        &= (1)(12.0107 amu) + (2)(15.9994 amu) \\
        &= 44.0095 amu \\
        &= 44.0095 \frac{g}{mol}                                
 \end{split}
 \end{equation}
\begin{equation} 
\begin{split}
    m &= n \times M \\
    n &= \frac{m}{M} \\
       &= \frac{88 g}{44.0095 \frac{g}{mol}} \\
       &= 1.999568 mol \\
       &= 2.0 mol                               
 \end{split}
 \end{equation}
 \end{solution}

\question[5] Jack knows that atomic mass of \ce{CO} is $28.0101 amu$.
What is the mass of one mole of \ce{CO}? 
\begin{solution}
\begin{equation} 
\begin{split}
    m &= n \times M \\
        &= (1 mol)(28.0101 \frac{g}{mol}) \\
        &= 28.0101 g                              
 \end{split}
 \end{equation}
 \end{solution}
